list / map (containers)
• list add ::xs 'item'.
• list remove ::xs 'item'.
• list length ::xs into @n
• list join ::xs with ', ' into @s
• map set ::m 'key' 'value'.
• map get ::m 'key' into @v
• map has ::m 'key' into @ok
• map merge ::a ::b into ::out
